NiFi User

NifiUser is the Schema for the nifi users API.

apiVersion: nifi.orange.com/v1alpha1
kind: NifiUser
metadata:
  name: example-user
  namespace: nifi
spec:
  clusterRef:
    name: nifi
  secretName: example-user-secret
  includeJKS: true

#NifiUser
